${text}
`; input.value = ""; try { const res = await fetch("https://ahmadzademohammadamin3-oss.github.io/AminGPT/", { method: "POST", headers: {"Content-Type": "application/json"}, body: JSON.stringify({"data":[text]}) }); const data = await res.json(); const reply = data.data[0]; chatBox.innerHTML += `
${reply}
`; chatBox.scrollTop = chatBox.scrollHeight; } catch (err) { console.error("خطا در ارسال پیام:", err); } }
