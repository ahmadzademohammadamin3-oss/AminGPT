async function send() { const input = document.getElementById("msg"); const text = input.value; if (!text) return; // اضافه کردن پیام کاربر const chatBox = document.getElementById("chatBox"); chatBox.innerHTML += `
${text}
`; input.value = ""; // پاک کردن باکس ورودی // ارسال پیام به Space try { const res = await fetch("https://ahmadzademohammadamin3-oss.github.io/AminGPT/", { method: "POST", headers: {"Content-Type": "application/json"}, body: JSON.stringify({"data":[text]}) }); const data = await res.json(); const reply = data.data[0]; // اضافه کردن پاسخ فقط **یک بار** chatBox.innerHTML += `
${reply}
`; // Scroll به پایین برای نمایش آخرین پیام chatBox.scrollTop = chatBox.scrollHeight; } catch (err) { console.error("خطا در ارسال پیام:", err); } }
